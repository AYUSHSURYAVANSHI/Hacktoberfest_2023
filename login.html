#include // Define a structure for a binary tree node struct Node { int data; Node* left; Node* right; Node(int value) : data(value), left(nullptr), right(nullptr) {} }; // Function to insert a new node into the binary tree Node* insert(Node* root, int value) { if (root == nullptr) { return new Node(value); } if (value < root->data) { root->left = insert(root->left, value); } else if (value > root->data) { root->right = insert(root->right, value); } return root; } // Function to print the binary tree in-order (left, root, right) void inOrderTraversal(Node* root) { if (root == nullptr) { return; } inOrderTraversal(root->left); std::cout << root->data << " "; inOrderTraversal(root->right); } int main() { Node* root = nullptr; // Insert some elements into the binary tree root = insert(root, 50); insert(root, 30); insert(root, 20); insert(root, 40); insert(root, 70); insert(root, 60); insert(root, 80); // Print the binary tree in-order std::cout << "In-order traversal of the binary tree: "; inOrderTraversal(root); std::cout << std::endl; return 0; }
